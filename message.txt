Napsal jsem krátký Python kód pro kontrolu schránky, který funguje:

from exchangelib import Credentials, Account, Configuration, DELEGATE
from exchangelib.protocol import BaseProtocol, NoVerifyHTTPAdapter
import warnings
from urllib3.exceptions import InsecureRequestWarning

warnings.simplefilter('ignore', InsecureRequestWarning) # vypni hlasky

# Ignorovat SSL certifikát
BaseProtocol.HTTP_ADAPTER_CLS = NoVerifyHTTPAdapter # nekontroluj certifikaty

#autentizace
creds = Credentials(username='WEFASINGEN\\obchodcz', password='T7bGCYYc91QPDC44') # tady je to dve lomitka a domena z nemecka WEFASINGEN 

# jinak ip je 10.0.0.220 - tak by na to mel byt ping
config = Configuration(
    service_endpoint='https://owa.wefa.com/EWS/Exchange.asmx', # Připojení pres login https://owa.wefa.com/ - tam máte jmeno a heslo
    credentials=creds,
    auth_type='NTLM' # musi NTLM 
)


account = Account(
    primary_smtp_address='obchodcz@wefa.com', # Připojení k mailboxu to je email 
    config=config,
    autodiscover=False,
    access_type=DELEGATE
)

# Stažení posledních 5 e-mailů - je tam ted 1 co jsem tam poslal
for item in account.inbox.all().order_by('-datetime_received')[:5]:
    print(item.datetime_received, item.subject)
